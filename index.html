CAPTCHA OCR Decoder

Upload a CAPTCHA image and extract its text using a Python OCR backend (pytesseract).

How it works

This page sends your uploaded image to a local Flask server that runs pytesseract (Tesseract OCR) to decode the text. Use the ‚ÄúEnhance pre-processing‚Äù option to binarize the image, which can help on noisy CAPTCHAs.

Backend (Python Flask) ‚Äî copy this into app.py 
# app.py
# Simple Flask backend for CAPTCHA OCR using pytesseract
# Requirements: pip install flask pillow pytesseract
# Also install Tesseract OCR binary on your system.

from flask import Flask, request, jsonify, send_from_directory
from PIL import Image
import pytesseract
import os

app = Flask(__name__)

# Optional: set pytesseract binary path via env var TESSERACT_CMD, e.g.:
# Windows example: set TESSERACT_CMD="C:\\Program Files\\Tesseract-OCR\\tesseract.exe"
# macOS/Linux example: export TESSERACT_CMD=/usr/local/bin/tesseract
tess_cmd = os.environ.get("TESSERACT_CMD")
if tess_cmd:
    pytesseract.pytesseract.tesseract_cmd = tess_cmd

@app.route("/")
def index():
    # Serve the front-end file from the same directory
    return send_from_directory(".", "index.html")

@app.route("/ocr", methods=["POST"])
def ocr():
    if "image" not in request.files:
        return jsonify({"error": "No image file uploaded (field name should be 'image')."}), 400

    file = request.files["image"]
    if file.filename == "":
        return jsonify({"error": "Empty filename."}), 400

    try:
        # Load image with PIL
        img = Image.open(file.stream).convert("L")  # grayscale
        # Optional pre-processing: simple binarization for noisy CAPTCHAs
        enhance = request.form.get("enhance") in ("1", "true", "on", "yes")
        if enhance:
            # Threshold to produce a clean black/white image
            img = img.point(lambda x: 0 if x < 140 else 255, "1")

        # Use PSM 7 (treat image as a single text line) for CAPTCHA-like images
        text = pytesseract.image_to_string(img, config="--psm 7")
        text = text.strip()

        return jsonify({"text": text})
    except Exception as e:
        return jsonify({"error": str(e)}), 500

if __name__ == "__main__":
    # Note: in production, use a proper WSGI server and disable debug.
    app.run(host="127.0.0.1", port=5000, debug=True)
